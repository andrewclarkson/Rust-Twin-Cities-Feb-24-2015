Functional Style

use std::iter;
use std::iter::AdditiveIterator;

let sum: u32 =
    iter::count(0, 1).
    map(|n| n * n).
    take_while(|&n| n < 1000u32).
    filter(|n| is_odd(*n)).
    sum();

fn is_odd(n: u32) -> bool { n % 2 == 1 }
